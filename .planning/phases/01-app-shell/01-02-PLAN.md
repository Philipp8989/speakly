```rust
// Speakly — Tauri App Setup (Phase 1)
// Initialisiert alle Plugins, den System Tray und das Close-to-Tray-Verhalten.

mod tray;

use tauri::{Manager, WindowEvent};
use tauri_plugin_store::StoreExt;
use serde_json::json;

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    tauri::Builder::default()
        // Plugin-Registrierung — Reihenfolge spielt keine Rolle
        .plugin(tauri_plugin_store::Builder::new().build())
        .plugin(tauri_plugin_positioner::init())
        .plugin(tauri_plugin_global_shortcut::Builder::new().build())
        .setup(|app| {
            // ConfigStore initialisieren — Standardwerte beim ersten Start setzen
            let store = app.store("settings.json")?;
            if store.get("initialized").is_none() {
                store.set("hotkey", json!("CommandOrControl+Shift+Space"));
                store.set("stt_mode", json!("cloud"));
                store.set("initialized", json!(true));
            }

            // System Tray aufbauen (Tray-Icon, Menues, Click-Handler)
            tray::setup_tray(app)?;

            // Close-to-Tray: Fenster schliessen verbirgt die App im Tray (per D-05, D-07)
            // Gilt fuer macOS UND Windows — app.exit(0) bleibt der einzige Weg zum Beenden
            let window = app.get_webview_window("main").unwrap();
            let win_clone = window.clone();
            window.on_window_event(move |event| {
                if let WindowEvent::CloseRequested { api, .. } = event {
                    api.prevent_close();
                    let _ = win_clone.hide();
                }
            });

            Ok(())
        })
        .run(tauri::generate_context!())
        .expect("Fehler beim Starten der Speakly-Anwendung");
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Tray menu → app.exit(0) | User-triggered quit from tray; should only terminate cleanly |
| WindowEvent::CloseRequested → hide | OS close event intercepted; must not block forced quit |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Denial of Service | prevent_close() intercept | mitigate | Exit path via app.exit(0) in tray menu (D-06) must remain intact; prevent_close() only hides, never permanently traps the app |
| T-02-02 | Tampering | settings.json on disk (plugin-store) | accept | Phase 1 only stores non-sensitive defaults (hotkey string, stt_mode, initialized flag); API keys not stored until Phase 2; file is in app data dir, not world-readable |
| T-02-03 | Spoofing | Tray icon identity | accept | No mitigable at app level; macOS notarization (Phase 7) + Windows signing (Phase 7) provide partial mitigation |
| T-02-04 | Elevation of Privilege | tauri-plugin-global-shortcut (no-op stub) | accept | Phase 1 registers hotkey with no handler; no command execution possible; real handler added in Phase 3 after AudioState design review |
</threat_model>
